SQL — Tabla de Chat
Ejecuta esto una vez en Supabase → SQL Editor para activar el chat.
create table if not exists public.client_messages (
  id bigint generated always as identity primary key,
  client_id uuid not null references auth.users(id) on delete cascade,
  coach_id  uuid not null references auth.users(id) on delete cascade,
  sender_id uuid not null references auth.users(id) on delete cascade,
  sender_role text not null check (sender_role in ('coach','client')),
  message text not null,
  created_at timestamptz not null default now()
);

alter table public.client_messages enable row level security;

-- Coach puede leer/escribir todos los mensajes de sus clientes
create policy "coach_rw" on public.client_messages
  for all using (coach_id = auth.uid());

-- Cliente solo ve sus propios mensajes
create policy "client_rw" on public.client_messages
  for all using (client_id = auth.uid());
